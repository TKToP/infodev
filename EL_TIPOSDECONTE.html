// Verificação de autenticação para conteúdo interativo
function checkContentAccess(user) {
  if (!user.authenticated || !user.subscriptionActive) {
    showPaywall();
    logAccessAttempt(user.id, 'DENIED');
    return false;
  }
  logAccessAttempt(user.id, 'GRANTED');
  return true;
}
